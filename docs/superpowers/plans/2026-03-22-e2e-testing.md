- [ ] **Step 1: Create e2e/helpers/crypto.ts**

This re-implements crypto.js logic using Node.js `crypto` module (for use in test helpers) and mirrors the Web Crypto API calls in the browser context.

```typescript
import { createCipheriv, createDecipheriv, randomBytes, pbkdf2Sync } from 'crypto';

const ITERATIONS = 100000;
const KEY_LENGTH = 32; // 256 bits for AES-256
const SALT_LENGTH = 16;
const IV_LENGTH = 12;
const TAG_LENGTH = 16;

function deriveKey(password: string, salt: Buffer): Buffer {
  return pbkdf2Sync(password, salt, ITERATIONS, KEY_LENGTH, 'sha256');
}

export async function encrypt(text: string, password: string): Promise<string> {
  const salt = randomBytes(SALT_LENGTH);
  const iv = randomBytes(IV_LENGTH);
  const key = deriveKey(password, salt);

  const cipher = createCipheriv('aes-256-gcm', key, iv);
  const encrypted = Buffer.concat([cipher.update(text, 'utf8'), cipher.final()]);
  const tag = cipher.getAuthTag();

  const combined = Buffer.concat([salt, iv, tag, encrypted]);
  return combined.toString('base64');
}

export async function decrypt(encryptedData: string, password: string): Promise<string> {
  const combined = Buffer.from(encryptedData, 'base64');

  const salt = combined.subarray(0, SALT_LENGTH);
  const iv = combined.subarray(SALT_LENGTH, SALT_LENGTH + IV_LENGTH);
  const tag = combined.subarray(SALT_LENGTH + IV_LENGTH, SALT_LENGTH + IV_LENGTH + TAG_LENGTH);
  const encrypted = combined.subarray(SALT_LENGTH + IV_LENGTH + TAG_LENGTH);

  const key = deriveKey(password, salt);

  const decipher = createDecipheriv('aes-256-gcm', key, iv);
  decipher.setAuthTag(tag);

  return decipher.update(encrypted).toString('utf8') + decipher.final('utf8');
}
```

- [ ] **Step 2: Create e2e/helpers/websocket.ts**

```typescript
import WebSocket from 'ws';

export interface WSMessage {
  type: string;
  [key: string]: any;
}

export class TestWebSocket {
  private ws: WebSocket | null = null;
  private messageHandlers: ((data: WSMessage) => void)[] = [];

  async connect(): Promise<void> {
    return new Promise((resolve, reject) => {
      this.ws = new WebSocket('ws://localhost:3000/ws');

      this.ws.on('open', () => resolve());
      this.ws.on('error', reject);
      this.ws.on('message', (data) => {
        const parsed = JSON.parse(data.toString());
        this.messageHandlers.forEach((h) => h(parsed));
      });
    });
  }

  send(msg: WSMessage): void {
    this.ws?.send(JSON.stringify(msg));
  }

  join(channel: string, username: string, password: string): Promise<WSMessage> {
    return this.waitForMessage('system', () => {
      this.send({ type: 'join', channel, username, password });
    });
  }

  sendMessage(text: string): Promise<WSMessage> {
    return this.waitForMessage('message', () => {
      this.send({ type: 'message', text });
    });
  }

  leave(): void {
    this.send({ type: 'leave' });
    this.ws?.close();
  }

  onMessage(handler: (data: WSMessage) => void): void {
    this.messageHandlers.push(handler);
  }

  private waitForMessage(
    expectedType: string,
    trigger: () => void
  ): Promise<WSMessage> {
    return new Promise((resolve) => {
      const handler = (data: WSMessage) => {
        if (data.type === expectedType) {
          this.messageHandlers = this.messageHandlers.filter((h) => h !== handler);
          resolve(data);
        }
      };
      this.messageHandlers.push(handler);
      trigger();
    });
  }
}
```

```typescript
import { Page, Locator } from '@playwright/test';

export class LoginPage {
  readonly page: Page;
  readonly usernameInput: Locator;
  readonly channelInput: Locator;
  readonly passwordInput: Locator;
  readonly submitButton: Locator;
  readonly errorMessage: Locator;

  constructor(page: Page) {
    this.page = page;
    this.usernameInput = page.locator('input[placeholder="请输入你的用户名"]');
    this.channelInput = page.locator('input[placeholder="输入频道名称，不存在则创建"]');
    this.passwordInput = page.locator('input[placeholder="输入频道密码"]');
    this.submitButton = page.locator('button[type="submit"]');
    this.errorMessage = page.locator(`.error`);
  }

  async login(username: string, channel: string, password: string): Promise<void> {
    await this.usernameInput.fill(username);
    await this.channelInput.fill(channel);
    await this.passwordInput.fill(password);
    await this.submitButton.click();
  }
}

export class ChatPage {
  readonly page: Page;
  readonly terminalTitle: Locator;
  readonly chatInput: Locator;
  readonly sendButton: Locator;
  readonly logoutButton: Locator;
  readonly messageList: Locator;

  constructor(page: Page) {
    this.page = page;
    this.terminalTitle = page.locator('.terminalTitle');
    this.chatInput = page.locator('textarea[placeholder*="输入消息"]');
    this.sendButton = page.locator('button:has-text("发送")');
    this.logoutButton = page.locator('button:has-text("登出")');
    this.messageList = page.locator('.chatMessages');
  }

  async sendMessage(text: string): Promise<void> {
    await this.chatInput.fill(text);
    await this.sendButton.click();
    // Wait for message to appear
    await this.page.waitForSelector(`.message:has-text("${text}")`, { timeout: 5000 });
  }

  async getLastMessage(): Promise<string> {
    const messages = await this.messageList.locator('.message').all();
    const last = messages[messages.length - 1];
    return last.textContent() || '';
  }
}
```

```typescript
import { test, expect, Page } from '@playwright/test';
import { TestWebSocket } from '../helpers/websocket';
import { ChatPage } from '../helpers/page-objects';

test.describe('Messaging', () => {

  let chatPage2: ChatPage;

  const PASSWORD = 'test-password-123';

    const context1 = await browser.newContext();
    const context2 = await browser.newContext();

    const page1 = await context1.newPage();
    const page2 = await context2.newPage();

    chatPage1 = new ChatPage(page1);
    chatPage2 = new ChatPage(page2);

    await page1.locator('input[placeholder="输入频道名称，不存在则创建"]').fill(CHANNEL);
    await page1.locator('input[placeholder="输入频道密码"]').fill(PASSWORD);
    await page1.locator('button[type="submit"]').click();
    await page1.waitForSelector('.terminalTitle');

    // Login user 2
    await page2.goto('/');
    await page2.locator('input[placeholder="请输入你的用户名"]').fill('User2');
    await page2.locator('input[placeholder="输入频道名称，不存在则创建"]').fill(CHANNEL);
    await page2.locator('input[placeholder="输入频道密码"]').fill(PASSWORD);
    await page2.locator('button[type="submit"]').click();
    await page2.waitForSelector('.terminalTitle');

    // Setup WebSocket connections
    ws1 = new TestWebSocket();
    ws2 = new TestWebSocket();
    await ws1.connect();
    await ws2.connect();
    await ws1.join(CHANNEL, 'User1', PASSWORD);
    await ws2.join(CHANNEL, 'User2', PASSWORD);

    // Wait for system messages to settle
    await page1.waitForTimeout(500);
    await page2.waitForTimeout(500);
  });

  test.afterEach(async () => {
    ws1?.leave();
    ws2?.leave();
  });

  test('should send and receive encrypted messages between users', async () => {
    // User 1 sends a message via UI
    const testMessage = 'Hello from User 1!';
    await chatPage1.sendMessage(testMessage);

    // User 2 should receive it (via WebSocket broadcast)
    await chatPage2.page.waitForSelector(
      `.message:has-text("${testMessage}")`,
      { timeout: 5000 }
    );

    const receivedMessage = await chatPage2.getLastMessage();
    expect(receivedMessage).toContain(testMessage);
  });

  test('should show correct sender for messages', async () => {
    const testMessage = 'Testing sender display';
    await chatPage1.sendMessage(testMessage);

    // User 2 should see it with User1's name
    await chatPage2.page.waitForSelector(

      { timeout: 5000 }

    await expect(messageElement.locator('.messageUser')).toContainText('User1');
```

```typescript
test.describe('Session Persistence & Lock Screen', () => {
  const CHANNEL = 'session-test-channel';

      return localStorage.getItem('terminal_chatroom_session');

    expect(sessionData!.length).toBeGreaterThan(0);
  });

  test('should show lock screen after inactivity', async ({ page }) => {
    await login(page, 'LockTestUser');

    // Simulate lock by waiting (LOCK_TIMEOUT is 5 minutes in production)

    // But a real test would wait

    // Just verify the chat interface is shown initially
    await expect(page.locator('.terminalTitle')).toBeVisible();
    await expect(page.locator('.chatInput')).toBeVisible();

      return localStorage.getItem('terminal_chatroom_session');

    // Should show login form
```

## Risk Assessment

| Risk | Likelihood | Impact | Mitigation |
|------|-------------|--------|------------|
| WebSocket timing issues in tests | Medium | Tests flaky | Use `waitForSelector` instead of fixed sleeps; generous timeouts |
| Crypto implementation mismatch | Low | Messages fail to decrypt | Test helper mirrors production crypto.js exactly |
| Port 3000 conflict | Low | Tests fail to start | Use `reuseExistingServer` in playwright.config |
| Session/localStorage state leaking between tests | Medium | Tests affect each other | Each test uses fresh browser context (`browser.newContext()`) |
| Playwright browser installation issues | Low | Cannot run tests | Use `npx playwright install --with-deps` |

## Notes

- Tests use two separate Playwright browser contexts to simulate two different users chatting simultaneously
- WebSocket helper uses Node.js `ws` library, while the frontend uses browser WebSocket API — both connect to the same `/ws` endpoint on the custom Next.js server
- The crypto helper re-implements the AES-256-GCM encryption from `crypto.js` using Node.js `crypto` module, ensuring test messages are encrypted the same way as real messages
- Playwright's `webServer` config starts `npm run dev` automatically before running tests, eliminating manual server startup
